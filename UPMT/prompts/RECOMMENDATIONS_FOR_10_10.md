def calculate_confidence(function, raw_data):
    """
    –û—Ü–µ–Ω–∏–≤–∞–µ—Ç —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å –≤ —Ç–æ–º —á—Ç–æ —Ñ—É–Ω–∫—Ü–∏—è —Ä–µ–∞–ª—å–Ω–æ –Ω—É–∂–Ω–∞.
    
    Returns: confidence_score (0.0-1.0)
    """
    score = 0.5  # baseline
    
    # +0.2: –£–ø–æ–º—è–Ω—É—Ç–∞ —è–≤–Ω–æ –≤ raw data
    if explicitly_mentioned(function, raw_data):
        score += 0.2
    
    # +0.1: –£–ø–æ–º—è–Ω—É—Ç–∞ –≤ –Ω–µ—Å–∫–æ–ª—å–∫–∏—Ö –º–µ—Å—Ç–∞—Ö (–ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ)
    mentions = count_mentions(function, raw_data)
    if mentions >= 3:
        score += 0.1
    
    # +0.1: –ï—Å—Ç—å –¥–µ—Ç–∞–ª–∏ —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–∏ (–Ω–µ –ø—Ä–æ—Å—Ç–æ —É–ø–æ–º–∏–Ω–∞–Ω–∏–µ)
    if has_implementation_details(function, raw_data):
        score += 0.1
    
    # +0.1: –°–≤—è–∑–∞–Ω–∞ —Å –¥—Ä—É–≥–∏–º–∏ —Ñ—É–Ω–∫—Ü–∏—è–º–∏ (–ª–æ–≥–∏—á–µ—Å–∫–∞—è —Å–≤—è–∑—å)
    if has_dependencies(function):
        score += 0.1
    
    # -0.2: Inferred (–Ω–µ —É–ø–æ–º—è–Ω—É—Ç–∞ —è–≤–Ω–æ, –≤—ã–≤–µ–¥–µ–Ω–∞ –ª–æ–≥–∏—á–µ—Å–∫–∏)
    if function.source == "inferred":
        score -= 0.2
    
    # -0.1: –ü—Ä–æ—Ç–∏–≤–æ—Ä–µ—á–∏–≤–∞—è –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—è
    if has_contradictions(function, raw_data):
        score -= 0.1
    
    return min(1.0, max(0.0, score))
